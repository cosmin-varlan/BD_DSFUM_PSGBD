void setup() {
  for(int i=8; i<=13; i++)
    pinMode(i,1);
}

int fr[6][2]={{13,100}, {12,130}, {11,400}, {10,200}, {9,250}, {8,1000}};

void loop() {
    for(int i=0; i<6; i++)
       if ((millis()/fr[i][1])%2) digitalWrite(fr[i][0],1);
                             else digitalWrite(fr[i][0],0);
}
